rcNcOpenGroup
int rcNcOpenGroup ( rcComm_t *  conn,
ncOpenInp_t *  ncOpenGroupInp,
int *  ncid 
)
Description:
Open a fully qualified group name and get the group id. On the server, nc_inq_grp_full_ncid is called to get the grpncid
Primary User:
client
Category:
NETCDF operations
Since:
1.0
Author:
Mike Wan
Date:
2012
Remarks:
none
Note:
none
Example Usage:

Get the group id in /myZone/home/john/myfile.nc:
ncOpenInp_t ncOpenInp;
int ncid = 0;
int grpncid = 0;
int status;
ncInqGrpsInp_t ncInqGrpsInp;
ncInqGrpsOut_t *ncInqGrpsOut = NULL;
ncInqInp_t ncInqInp;
ncInqOut_t *ncInqOut = NULL;
bzero (&ncOpenInp, sizeof (ncOpenInp));
rstrcpy (ncOpenInp.objPath, "/myZone/home/john/myfile.nc", MAX_NAME_LEN);
ncOpenInp.mode = NC_NOWRITE;
status = rcNcOpen (conn, &ncOpenInp, &ncid);
if (status < 0) {
.... handle the error
}
bzero (&ncInqGrpsInp, sizeof (ncInqGrpsInp));
ncInqGrpsInp.ncid = ncid;
status = rcNcInqGrps (conn, &ncInqGrpsInp, &ncInqGrpsOut);
if (status < 0) {
.... handle the error
}
if (ncInqGrpsOut->ngrps <= 0) {
.... handle the error
}
rstrcpy (ncOpenInp.objPath, ncInqGrpsOut->grpName[0], MAX_NAME_LEN);
freeNcInqGrpsOut (&ncInqGrpsOut);
ncOpenInp.rootNcid = ncid;
status = rcNcOpenGroup (conn, &ncOpenInp, &grpncid);
if (status < 0) {
.... handle the error
}
// do the general inq on the group
bzero (&ncInqInp, sizeof (ncInqInp));
ncInqInp.ncid = grpncid;
ncInqInp.paramType = NC_ALL_TYPE;
ncInqInp.flags = NC_ALL_FLAG;
status = rcNcInq (conn, &ncInqInp, &ncInqOut);
if (status < 0) {
.... handle the error
}

Parameters:
[in]conn- A rcComm_t connection handle to the server.
[in]ncOpenInp- Elements of ncOpenInp_t used :
  • char objPath[MAX_NAME_LEN] - full path of the data object.
  • int mode - the mode of the open - valid values are given in netcdf.h - NC_NOWRITE, NC_WRITE
[out]ncid- the ncid of the opened object.
Returns:
integer
