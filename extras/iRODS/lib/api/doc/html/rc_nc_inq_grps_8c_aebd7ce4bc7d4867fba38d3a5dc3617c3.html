rcNcInqGrps
int rcNcInqGrps ( rcComm_t *  conn,
ncInqGrpsInp_t *  ncInqGrpsInp,
ncInqGrpsOut_t **  ncInqGrpsOut 
)
Description:
Given the ncid, returns all full group paths. On the server, the nc_inq_grpname_len and nc_inq_grpname_full are called
Primary User:
client
Category:
NETCDF operations
Since:
3.1
Author:
Mike Wan
Date:
2012
Remarks:
none
Note:
none
Example Usage:

Get the full group path in /myZone/home/john/myfile.nc:
ncOpenInp_t ncOpenInp;
int ncid = 0;
int grpncid = 0;
int status;
ncInqGrpsInp_t ncInqGrpsInp;
ncInqGrpsOut_t *ncInqGrpsOut = NULL;
ncInqInp_t ncInqInp;
ncInqOut_t *ncInqOut = NULL;
bzero (&ncOpenInp, sizeof (ncOpenInp));
rstrcpy (ncOpenInp.objPath, "/myZone/home/john/myfile.nc", MAX_NAME_LEN);
ncOpenInp.mode = NC_NOWRITE;
status = rcNcOpen (conn, &ncOpenInp, &ncid);
if (status < 0) {
.... handle the error
}
bzero (&ncInqGrpsInp, sizeof (ncInqGrpsInp));
ncInqGrpsInp.ncid = ncid;
status = rcNcInqGrps (conn, &ncInqGrpsInp, &ncInqGrpsOut);
if (status < 0) {
.... handle the error
}
if (ncInqGrpsOut->ngrps <= 0) {
.... handle the error
}
rstrcpy (ncOpenInp.objPath, ncInqGrpsOut->grpName[0], MAX_NAME_LEN);
freeNcInqGrpsOut (&ncInqGrpsOut);
ncOpenInp.rootNcid = ncid;
status = rcNcOpenGroup (conn, &ncOpenInp, &grpncid);
if (status < 0) {
.... handle the error
}

Parameters:
[in]conn- A rcComm_t connection handle to the server.
[in]ncInqGrpsInp- Elements of ncInqGrpsInp_t used :
  • int ncid - the ncid of the opened object from rcNcOpen.
[out]ncInqGrpsOut- Elements of ncInqGrpsOut_t:
  • int ngrps - number of groups
  • char **grpName - An array of ngrps pointers containing the group names.
