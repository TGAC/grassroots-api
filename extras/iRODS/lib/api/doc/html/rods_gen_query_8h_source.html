00018 /* In genQueryInp_t, selectInp is a int index, int value pair. The index
00019  * represents the attribute index. 
00020  * sqlCondInp is a int index, string value pair. The index
00021  * represents the attribute index. 
00022  */
00023 
00024 typedef struct GenQueryInp {
00025     int maxRows;             /* max number of rows to return, if 0 
00026                                 close out the SQL statement call (i.e. instead
00027                                 of getting more rows until it is finished). */
00028     int continueInx;         /* if non-zero, this is the value returned in
00029                                 the genQueryOut structure and the current
00030                                 call is to get more rows.  In this case, the
00031                                 selectInp & sqlCondInp arguments are ignored.*/
00032     int rowOffset;           /* if positive, return rows starting with
00033                                 this index (skip earlier ones), 0-origin  */
00034     int options;             /* Bits for special options, currently:
00035                                 If RETURN_TOTAL_ROW_COUNT is set, the total
00036                                 number of available rows will be returned
00037                                 in totalRowCount (causes a little overhead
00038                                 so only request it if needed).  If rowOffset
00039                                 is also used, totalRowCount will include
00040                                 the skipped rows.
00041                                 If NO_DISTINCT is set, the normal 'distinct'
00042                                 keyword is not included in the SQL query.
00043                                 If QUOTA_QUERY is set, do the special quota
00044                                 query.
00045                                 If AUTO_CLOSE is set, close out the statement
00046                                 even if more rows are available.  -1 is
00047                                 returned as the continueInx if there were
00048                                 (possibly) additional rows available.
00049                 If UPPER_CASE_WHERE is set, make the 'where'
00050                                 columns upper case.
00051                              */
00052     keyValPair_t condInput;
00053     inxIvalPair_t selectInp; /* 1st int array is columns to return (select),
00054                                 2nd int array has bits for special options:
00055                                 currently ORDER_BY and ORDER_BY_DESC */
00056     inxValPair_t sqlCondInp; /* 1st array is columns for conditions (where),
00057                                 2nd array has strings for the conditions. */
00058 } genQueryInp_t;

00079 primary ordering column.

00093 /* 
00094   These are some operations (functions) that can be applied to columns
00095   being returned (selected) by setting the input array,
00096   genQueryInp.selectInp.value[i], to these values.  Values 0 and 1 (or
00097   any not defined here) will just return the column content (i.e. the
00098   normal, default case).  The ORDER_BY and ORDER_BY_DESC bits are also
00099   stored in the same input array but are not normally used together
00100   with these since these will return one row.
00101 */
